ImageFlipbook - Basic Example

A simple demonstration of the ImageFlipbook library

HTML:

<div id="my-flipbook"></div>

JavaScript:

const flipbook = new ImageFlipbook({
    container: '#my-flipbook',
    title: 'My Image Collection',
    pages: [
        'https://picsum.photos/800/600?random=1',
        'https://picsum.photos/800/600?random=2',
        'https://picsum.photos/800/600?random=3',
        'https://picsum.photos/800/600?random=4',
        'https://picsum.photos/800/600?random=5'
    ],
    chapters: [
        { name: "Start", page: 1, icon: "📖" },
        { name: "Middle", page: 3, icon: "📄" },
        { name: "End", page: 5, icon: "🔚" }
    ],
    theme: 'default',
    fullscreenEnabled: true,
    socialSharing: true
});
